  5 /**
  6  * @memberof neural
  7  * The store of transfer functions which can be used in Neurons to convert input sums to activation values.  By default, the following are available:
  8  * transferFunctions.logSigmoid - see {@link https://en.wikipedia.org/wiki/Logistic_function}
  9  * transferFunctions.rectifier - see {@link https://en.wikipedia.org/wiki/Rectifier_(neural_networks)}
 10  * transferFunctions.linear - useful for output layers for training data which could take any value.  Note that neural networks cannot rely solely on linear transfer functions otherwise hidden layers will be effectively redundant.
 11  */
 12 var transferFunctions = {}

 14 /**
 15  * Creates a new Neuron
 16  * @constructor
 17  * @memberof neural
 18  * @param {Object} params
 19  * @param {fn} [params.transfer='logSigmoid'] The neuron's transfer function.
 20  * @param {Array=} [params.inputs] An array of inputs to the neuron, which can be numbers, functions returning a number or Neurons
 21  * @param {Array=} [params.weights] The input weights with which to initialise the Neuron.  Defaults to zeros.
 22  */
 23 function Neuron (params) {
 24   if (!(this instanceof Neuron)) return new Neuron(params)
 25   var transfer = params.transfer || transferFunctions.logSigmoid
 26   var inputs = params.inputs || []
 27   var weights = params.weights || matrix.zeros(inputs.length)
 28   var outputNeurons = []
 29   var expectedOutputVal = 0
 30   var activationCache = null
 31   var inputSumCache = null
 32   var delta = 0
 33   var id = (neuronId += 1)
 34 
 35   /**
 36    * Resets the inputs to a neuron.  Any existing inputs are removed.
 37    * @param  {Array} newInputs Array of inputs to the Neuron, which can be numbers, functions returning a number or Neurons.
 38    */
 39   this.setInputs = function (newInputs) {
 40     if (newInputs.length !== inputs.length) weights = matrix.zeros(newInputs.length)
 41     this.invalidate()
 42     inputs = newInputs
 43   }
 44 
 45   /**
 46    * Returns the Neurons inputs.
 47    * @return {Array}  The Neurons inputs, which could be numbers, functions returning a number or Neurons.
 48    */
 49   this.getInputs = function () {
 50     return inputs
 51   }
 52 
 53   /**
 54    * Returns the number of inputs feeding this neuron.
 55    * @return {number}  count of inputs, which could be numbers, functions returning a number of Neurons.
 56    */
 57   this.getInputCount = function () {
 58     return inputs.length
 59   }
 60 
 61   /**
 62    * Updates the Neuron's weights.
 63    * @param  {Array} newWeights An array of numbers, which must be of the same length as the Neuron's array of inputs.
 64    */
 65   this.updateWeights = function (newWeights) {
 66     this.invalidate()
 67     weights = newWeights.slice(0)
 68   }
 69 
 70   /**
 71    * Returns the Neuron's weights.
 72    * @return {Array}  The neuron's weights (numbers).
 73    */
 74   this.getWeights = function () {
 75     return weights.slice(0)
 76   }
 77 
 78   /**
 79    * Returns the Neuron's id.
 80    * @return {number}  The Neuron's id, which is unique at the module level.
 81    */
 82   this.getId = function () {
 83     return id
 84   }
 85 
 86   /**
 87    * Get the Neuron's transfer function
 88    * @return {fn}  The transfer function
 89    */
 90   this.getTransfer = function () {
 91     return transfer
 92   }

 94   /**
 95    * Sets the Neuron's transfer function
 96    * @param  {fn} The replacement transfer function (see section on transfer functions for details of acceptable formats).
 97    */
 98   this.setTransfer = function (fn) {
 99     transfer = fn
100   }

102   /**
103    * Gets the Neuron's current activation value.  Note that this does NOT recalculate the value if input values or weights have changed.
104    * @return {number}  The most recently-calculated activation value.
105    */
106   this.getActivation = function () {
107     return activationCache
108   }
109 
110   /**
111    * Gets the Neuron's current input sum.  This is the weighted sum of inputs, prior to having been passed through the transfer function. Note that this does NOT recalculate the value if input values or weights have changed.
112    * @return {number}  The most recently-calculated input sum.
113    */
114   this.getInputSum = function () {
115     return inputSumCache
116   }
117 
118   /**
119    * Gets the Neuron's current delta.
120    * @return {number}  The Neuron's delta.
121    */
122   this.getDelta = function () {
123     return delta
124   }
125 
126   /**
127    * Indicates whether the Neuron is in the output layer of a Network
128    * @return {boolean}
129    */
130   this.isOutput = function () {
131     return expectedOutputVal !== null
132   }
133 
134   /**
135    * Sets the expected activation value (from training data) of the Neuron
136    * @param  {number}  Expected output value
137    */
138   this.setExpected = function (val) {
139     expectedOutputVal = val
140   }
141 
142   this._updateOutputNeurons = function (neurons) {
143     expectedOutputVal = null
144     outputNeurons = neurons
145   }
146 
147   /**
148    * Plugs in a new input to the Neuron, which is initialised with a respective weight of 0.
149    * @param  {number, fn or Neuron} The new input.
150    */
151   this.addInput = function (input) {
152     inputs.push(input)
153     weights.push(0)
154     this.invalidate()
155   }
156 
157   /**
158    * Recalculates the activation value, recalculating all the input nodes in turn as required.
159    * @return {number}  The recalculated activation value.
160    */
161   this.calc = function () {
162     if (activationCache !== null) return activationCache
163     inputSumCache = matrix.mult(matrix.invoke(inputs), matrix.transpose(weights), true)
164     activationCache = transfer(inputSumCache)
165 
166     return activationCache
167   }
168 
169   /**
170    * The difference between the most recently calculated activation value and the last expected value to be supplied.
171    * @return {number}  Difference between activation and expected values.
172    */
173   this.error = function () {
174     return activationCache - expectedOutputVal
175   }
176 
177   /**
178    * Marks the Neuron as in need of recalculation by clearing the activation value cache.
179    */
180   this.invalidate = function () {
181     activationCache = null
182   }
183 
184   /**
185    * Recalculates the Neuron's delta.
186    * @return {number}  The new delta.
187    */
188   this.calcDelta = function () {
189     if (this.isOutput()) return this._calcDeltaOutputLayer()
190     delta = outputNeurons.reduce((total, neuron) => {
191       var index = neuron.getInputs().findIndex(input => input.getId && input.getId() === id)
192       return total + (neuron.getDelta() * neuron.getWeights()[index])
193     }, 0) * transfer.derivative(inputSumCache, activationCache)
194     check(delta, 'Delta non-output layer')
195     return delta
196   }
197 
198   this._calcDeltaOutputLayer = function () {
199     delta = -(expectedOutputVal - activationCache) * transfer.derivative(inputSumCache, activationCache)
200     check.call(this, expectedOutputVal, 'expectedOutputVal')
201     check.call(this, activationCache, 'activationCache')
202     check.call(this, inputSumCache, 'inputSumCache')
203     return delta
204   }
205 
206   /**
207    * Returns the partial derivative of the Neuron's error with respect to its output weights.
208    * @return {Array}     The output weight partials.
209    */
210    this.getOutputWeightPartials = function () {
211     return outputNeurons.map(neuron => activationCache * neuron.getDelta())
212   }
213 
214   /**
215    * Returns the partial derivatives of the input Neurons errors with respect to this Neuron's input weights.
216    * @return {Array}  The input weight partials.
217    */
218   this.getInputWeightPartials = function () {
219     return inputs.map(input => (input instanceof Neuron) ? input.getActivation() * delta : delta)
220   }
221 
222   /**
223    * Randomizes the Neuron's input weights.
224    * @param  {number} [max=0.000001] Maximum possible weight.
225    * @param  {number} [min=0] Minimum possible weights.
226    */
227   this.randomizeWeights = function (max, min) {
228     return weights.map((weight, ind) => {
229       var newWeight = randNum(max || 0.000001, min)
230       weights[ind] = newWeight
231     })
232   }
233 
234   return this
235 }

237 /**
238  * Creates a new Network Layer
239  * @constructor
240  * @memberof neural
241  * @param {Object} params
242  * @param {number or Array} [params.neurons]  The Neurons in the layer. If an integer is passed, that number of Neurons will be constructed for this Layer.
243  */
244 function Layer (params) {
245   var neurons = []
246   var isOutput = true
247   var isInput = true
248 
249   if (!(this instanceof Layer)) return new Layer(params)
250   if (params.neurons instanceof Array) {
251     neurons = params.neurons
252   }
253   if (typeof params.neurons === 'number') {
254     neurons = Array(params.neurons).fill(0).map(() => {
255       return new Neuron(params)
256     })
257   }
258 
259   /**
260    * Returns whether this an output layer.  Note that all layers are output layers by default, until they have their neurons plugged into another layer.
261    * @return {boolean}
262    */
263   this.isOutput = function () {
264     return isOutput
265   }
266 
267   /**
268    * Overrides the layer's status.
269    * @param  {Object} statuses
270    * @param  {boolean} statuses.input Whether the layer should be marked as an input layer.
271    * @param  {boolean} statuses.output Whether the layer should be marked as an output layer.
272    */
273   this.setStatus = function (statuses) {
274     if (typeof statuses.input === 'boolean') isInput = statuses.isInput
275     if (typeof statuses.output === 'boolean') isOutput = statuses.isOutput
276   }
277 
278   /**
279    * Plugs this Layer into another Layer such that the Neurons in this Layer become input Neurons for the supplied Layer.
280    * @param  {Layer} outputLayer The Layer which will have its Neurons take the Neurons in this Layer as inputs.
281    * @return {Layer}             The supplied output Layer for chaining purposes.
282    */
283   this.plug = function (outputLayer) {
284     // Add an extra input for bias
285     outputLayer.setStatus({ input: false })
286     outputLayer.setInputs(neurons.concat(1))
287     isOutput = false
288     neurons.forEach(neuron => {
289       neuron._updateOutputNeurons(outputLayer.getNeurons())
290     })
291     return outputLayer
292   }
293 
294   /**
295    * Returns the Neurons which make up this layer.
296    * @return {Array}
297    */
298   this.getNeurons = function () {
299     return neurons
300   }
301 
302   /**
303    * Sets the inputs for all the Neurons in this Layer.
304    * @param  {Array} inputs       An array of inputs, which could be numbers, functions returning a number or Neurons.
305    * @param  {boolean} isInputLayer Indicates whether this is intended to be the input layer in a Network.  If set to true, rather than each of the supplied inputs being wired into each of the Neurons in this Layer, they will be mapped one-to-one, with no bias (i.e. passed straight through to the next layer).
306    */
307   this.setInputs = function (inputs, isInputLayer) {
308     if (!(inputs instanceof Array)) throw new Error('Inputs must be an array')
309     if (isInputLayer) {
310       if (inputs.length !== neurons.length) throw new Error(`Number of inputs (${inputs.length}) differs from number of neurons in input layer (${neurons.length})`)
311       isInput = true
312       neurons.forEach((neuron, ind) => {
313         neuron.setInputs([inputs[ind]])
314         neuron.updateWeights([1])
315       })
316       this.getNeurons().forEach(neuron => neuron.setTransfer(transferFunctions.linear))
317     } else {
318       if (inputs.some(input => !(input instanceof Array))) {
319         inputs = Array(neurons.length).fill(0).map(() => {
320           return inputs.slice(0)
321         })
322       }
323       inputs.forEach((inputArray, ind) => {
324         neurons[ind].setInputs(inputArray)
325       })
326     }
327   }
328 
329   /**
330    * Sets the expected values for the activation values of the Neurons in this Layer.
331    * @param  {Array} An array of output values (numbers), equal in length to the number of Neurons in this Layer.
332    */
333   this.setExpected = function (outputs) {
334     if (!(outputs instanceof Array)) throw new Error('Outputs must be an array')
335     if (outputs.length !== neurons.length) throw new Error(`Number of outputs (${outputs.length}) differs from number of neurons in layer (${neurons.length})`)
336     neurons.forEach((neuron, ind) => neuron.setExpected(outputs[ind]))
337   }
338 
339   /**
340    * Sets the transfer function for every Neuron in this Layer
341    * @param  {fn} fn The replacement transfer function (see section on transfer functions for details of acceptable formats).
342    */
343   this.setTransfer = function (fn) {
344     return neurons.forEach(neuron => neuron.setTransfer(fn))
345   }
346 
347   /**
348    * Recalculates the activation values for all of the Neurons in this Layer, recalculating their input Neuron values in sequence as required.
349    * @return {Array}  The recalculated activation values.
350    */
351   this.calc = function () {
352     return neurons.map(neuron => neuron.calc())
353   }
354 
355   /**
356    * Invalidates the activation cache for all the Neurons in this Layer.
357    */
358   this.invalidate = function () {
359     neurons.map(neuron => neuron.invalidate())
360   }
361 
362   /**
363    * Recalculates the deltas for all Neurons in this Layer.

365    */
366   this.calcDeltas = function () {

368   }

374   this.getWeights = function () {
375     return neurons.map(neuron => neuron.getWeights())
376   }
377

380    * @param  {Array of Arrays} weights The new weights for each Neuron in this Layer.

382   this.setWeights = function (weights) {
383     neurons.forEach((neuron, neuronInd) => neuron.updateWeights(weights[neuronInd]))
384   }
385

387    * Returns the activation values for the Neurons in this Layer.
388    * @return {Array} The activation values.
389    */
390   this.getActivations = function () {
391     return neurons.map(neuron => neuron.getActivation())
392   }

394   /**
395    * Returns the input sums for the Neurons in this Layer (the weighted sums of inputs for each Neuron before they've been passed through the transfer function).
396    * @return {Array} The input sums.
397    */
398   this.getInputSums = function () {
399     return neurons.map(neuron => neuron.getInputSum())
400   }
401 
402   /**
403    * Returns the ids for the Neurons in this Layer.
404    * @return {Array}  An array of ids.
405    */
406   this.getIds = function () {
407     return neurons.map(neuron => neuron.getId())
408   }
409 
410   /**
411    * Returns the output weight partials for the Neurons in this Layer.
412    * @see {@link Neuron#getOutputWeightPartials}
413    * @return {Array of Arrays}  Array of output weight partials for each Neuron.
414    */
415   this.getOutputWeightPartials = function () {
416     return neurons.map(neuron => neuron.getOutputWeightPartials())
417   }
418 
419   /**
420    * Returns the input weight partials for the Neurons in this Layer.
421    * @see {@link Neuron#getInputWeightPartials}
422    * @return {Array of Arrays}  Array of input weight partials for each Neuron.
423    */
424   this.getInputWeightPartials = function () {
425     return neurons.map(neuron => neuron.getInputWeightPartials())
426   }

428   /**
429    * Randomizes the weights for all of the Neurons in this Layer.
430    * @param  {number} e The randomized weights will be in the range [-e, e].
431    */
432   this.randomizeWeights = function (e) {
433     if (isInput) return

440 /**
441  * Creates a new Network
442  * @constructor
443  * @memberof neural
444  * @param {Object} params
445  * @param {Array} [params.layers] An array of layer sizes, indicating the number of Neurons in each Layer (and implicitly, the number of Layers).
446  */
447 function Network (params) {
448   if (!(this instanceof Network)) return new Network(params)
449   var layers = []
450   this.id = randomId()
451 
452   if (!(params.layers instanceof Array)) throw new Error('You must supply a "layers" parameter which is an array')
453   params.layers.forEach(layer => {
454     if (layer instanceof Layer) {
455       layers.push(layer)
456     } else if (typeof layer === 'number') {
457       var layerParams = Object.assign({}, params, { neurons: layer })
458       layers.push(new Layer(layerParams))
459     } else {
460       throw new Error('Unrecognised layer entry')
461     }
462   })
463 
464   layers.reduce((lastLayer, thisLayer) => {
465     if (lastLayer) lastLayer.plug(thisLayer)
466     return thisLayer
467   }, null)
468 
469   /**
470    * Returns the number of layers in the network.
471    * @return {number}
472    */
473   this.layerCount = function () {
474     return layers.length
475   }
476 
477   /**
478    * Returns the Layer instances which make up the network (from input to output).
479    * @return {Array}  Array of Layers.
480    */
481   this.getLayers = function () {
482     return layers
483   }
484 
485   /**
486    * Returns the Network's input Layer.
487    * @return {Layer}
488    */
489   this.inputLayer = function () {
490     return layers[0]
491   }
492 
493   /**
494    * Returns the Network's output Layer.
495    * @return {Layer}
496    */
497   this.outputLayer = function () {
498     return layers[layers.length - 1]
499   }
500 
501   /**
502    * Sets the inputs for the Network's input Layer.
503    * @see {@link Layer#setInputs} for more details.
504    * @param  {Array} inputs An array of inputs, which must be of the same length as the number of Neurons in the input Layer.  Note that whilst these could be Neurons, in an input Layer they would more normally be numbers or functions.
505    */
506   this.setInputs = function (inputs) {
507     this.inputLayer().setInputs(inputs, true)
508   }
509 
510   /**
511    * Sets the expected outputs for the Network's output Layer.
512    * @see {@link Layer#setOutputs} for more details.
513    * @param  {Array} outputs An array of expected values for the output Layer.  This must be the same length as the number of Neurons in the output Layer.
514    */
515   this.setExpected = function (outputs) {
516     this.outputLayer().setExpected(outputs)
517   }
518 
519   /**
520    * Calculates the output values for the Network based on the current inputs using forward propagation. Note that if Neuron activation values have been calculated since the previous invalidation, these will be used rather than recalculation occurring.
521    * @return {Array}             The output layer activation values resulting from the current network inputs.
522    */
523   this.calc = function () {
524     return this.outputLayer().calc()
525   }
526 
527   /**
528    * Gets all activation values for all Neurons in all Layers in the Network.
529    * @return {Array of Arrays} Activation values.
530    */
531   this.getActivations = function () {
532     return layers.map(layer => layer.getActivations())
533   }
534 
535   /**
536    * Gets all input sums for all Neurons in all Layers in the Network.
537    * @see {@link Neuron#getInputSum} for more details.
538    * @return {Array of Arrays} Input sums.
539    */
540   this.getInputSums = function () {
541     return layers.map(layer => layer.getInputSums())
542   }
543 
544   /**
545    * Marks the activation cache for every Neuron in the Network as invalid.
546    */
547   this.invalidate = function () {
548     layers.forEach(layer => layer.invalidate())
549   }
550 
551   /**
552    * Randomizes the input weights for all the Neurons in the Network.
553    * @see {@link Layer#randomizeWeights} for more details.
554    * @param  {type} e Resulting weights will be in the interval [-e, e].
555    */
556   this.randomizeWeights = function (e) {
557     layers.forEach(layer => layer.randomizeWeights(e))
558   }
559 
560   /**
561    * Peforms a full forward propagation of the Network using the supplied input values.  Also optionally sets the expected output values for error calculation and training.
562    * @param  {Object} trial
563    * @param  {Array} [trial.inputs] An array of input values to feed into the Network's input Layer.  This must be the same length as the number of Neurons in the input Layer.
564    * @param  {Array} [trial.outputs=] An array of output values to mark as the output Layer's expected activation values.  This must be the same length as the number of Neurons in the output Layer.
565    * @return {Array}      The actual output values resulting from the supplied inputs with the current network weights.
566    */
567   this.forwardPropagate = function (trial) {
568     this.invalidate()
569     this.setInputs(trial.inputs)
570     this.setExpected(trial.outputs)
571     return this.calc()
572   }
573 
574   /**
575    * Returns the sum-squared error resulting from comparing the calculated Network outputs with the expected outputs, using the current inputs and weights.
576    * @return {number}
577    */
578   this.sumSqError = function () {
579     return 0.5 * this.outputLayer().getNeurons().reduce((sum, neuron) => {
580       return sum + Math.pow(neuron.error(), 2)
581     }, 0)
582   }
583 
584   /**
585    * Applies the back-propagation algorithm to recalculate the deltas for each Neuron in the Network, working from the output Layer to the input Layer.
586    * @return {Array of Arrays}  The deltas for the Neurons in each of the output Layers.
587    */
588   this.backPropagate = function () {
589     this.invalidate()
590     this.calc()
591     var layerCount = this.layerCount()
592     return layers.map((ignore, ind) => {
593       var layer = layers[layerCount - ind - 1]
594       return layer.calcDeltas()
595     })
596   }
597 
598   /**
599    * Returns the ids for all the Neurons in the Network.
600    * @return {Array of Arrays}
601    */
602   this.getIds = function () {
603     return layers.map(layer => layer.getIds())
604   }
605 
606   /**
607    * Gets the output weight partials for each Neuron in the Network
608    * @see {@link Neuron#getInputWeightPartials}
609    * @return {Array of Array of Arrays}  Input weight partials for each Neuron.
610    */
611   this.getOutputWeightPartials = function () {
612     return layers.map(layer => layer.getOutputWeightPartials())
613   }
614 
615   /**
616    * Gets the output weight partials for each Neuron in the Network
617    * @see {@link Neuron#getOutputWeightPartials}
618    * @return {Array of Array of Arrays}  Output weight partials for each Neuron.
619    */
620   this.getInputWeightPartials = function () {
621     return layers.map(layer => layer.getInputWeightPartials())
622   }
623 
624   /**
625    * Gets the input weights for each Neuron in the network.
626    * @return {Array of Array of Arrays}  Input weights by input, Neuron and Layer.
627    */
628   this.getWeights = function () {
629     return layers.map(layer => layer.getWeights())
630   }
631 
632   /**
633    * Sets the weights for the entire network in one go.  Useful for rebuilding a trained network.
634    * @param  {Array of Array of Arrays} The inner-most Arrays refer to the input weights for each Neuron.  These should be arranged in Arrays corresponding to each Neuron in a Layer.  Finally, the weights for each Layer should make up the outer-most Array.
635    */
636   this.setWeights = function (weights) {
637     layers.map((layer, layerInd) => layer.setWeights(weights[layerInd].slice(0)))
638   }
639 
640   return this
641 }

680 /**
681 * @memberof neural
682  * Returns a generator which trains the given Network with one epoch of the supplied training data (i.e. one update for each example in the training set) when `next` is invoked.
683  * Calling `next` will return an object with the current epoch, average sum-squared error of the last epoch, the best achieved error, the associated best network weights and the current learning rate (alpha).
684  * @param  {Network} network      The neural network to train.
685  * @param  {TrainingData} trainingData A TrainingData object to use to train the network.
686  * @param  {Object} params
687  * @param  {number} [params.alpha=0.5] The initial learning rate.
688  * @param  {number} [params.lambda=0.01] The damping factor (which pulls weights back towards 0 to avoid them escalating).
689  * @param  {number} [params.progressiveAlpha=] Allows the learning rate to increase gradually when calculated errors are declining, and pull back if they increase.
690  * @param  {number} [params.progressiveAlpha.creep=1.01] The proportional increase in alpha (learning rate) after every epoch which results in an improved net error.
691  * @param  {number} [params.progressiveAlpha.reversal=0.75] The pull-back in alpha (learning rate) after any epoch which results in an increased net error.
692  * @param  {number} [params.progressiveAlpha.floor=0] The lowest level the learning rate will be permitted to reach, even if errors continue to increase. Can help to avoid problems with local minima.
693  * @param  {boolean} [params.verbose=false] Also yield the calculated (actual) outputs for the last training epoch, for visualisation of training progress.
694  * @return {generator}
695  */
696 function* trainer (network, trainingData, params) {
697   params = params || {}
698   var alpha = params.alpha || network.alpha || 0.5
699   var alphaUpdater = alphaUpdaterGen(alpha)
700   var lambda = params.lamdba || network.lambda || 0.01
701   var dataLength = trainingData.dataLength()
702   var progressiveAlpha = params.progressiveAlpha ? {
703     creep: params.progressiveAlpha.creep || 1.01,
704     reversal: params.progressiveAlpha.reversal || 0.75,
705     floor: params.progressiveAlpha.floor || 0
706   } : false
707   var verbose = params.verbose
708   var minError
709   var bestWeights
710   var epoch = 0
711   var complete = false
712 
713   network.randomizeWeights(0.00000001)
714   while (!complete) {
715     complete = yield runEpoch()
716   }
717 
718   function runEpoch () {
719     epoch += 1
720     var dataGen = trainingData.dataGenerator()
721     var error = 0
722     var outputs = []
723 
724     for (var trial of dataGen) {
725       error += feedForwardAndCalcError(trial)
726       network.backPropagate()
727       if (verbose) {
728         outputs.push(network.outputLayer().getActivations())
729       }
730       updateNetworkWeights(network.getInputWeightPartials())
731     }
732 
733     var weightedError = Math.pow(error / dataLength, 0.5)
734     if (progressiveAlpha) alpha = alphaUpdater.next(weightedError).value
735     if (!minError || weightedError < minError) {
736       minError = weightedError
737       bestWeights = network.getWeights()
738     }
739     var result = {
740       epoch: epoch,
741       error: weightedError,
742       minError: minError,
743       bestWeights: bestWeights,
744       alpha: alpha
745     }
746     if (verbose) result.outputs = outputs
747     return result
748   }
749 
750   function feedForwardAndCalcError (trial) {
751     network.forwardPropagate(trial)
752     return network.sumSqError()
753   }
754 
755   function updateNetworkWeights (weightMap) {
756     var layers = network.getLayers()
757     layers.forEach((layer, layerInd) => {
758       if (layerInd === 0) return
759       layer.getNeurons().forEach((neuron, neuronInd) => {
760         var neuronWeights = neuron.getWeights()
761         var weightMapThisNeuron = weightMap[layerInd][neuronInd]
762         neuronWeights = neuronWeights.map((neuronWeight, neuronWeightInd) => {
763           return neuronWeight - (alpha * (weightMapThisNeuron[neuronWeightInd] + (lambda * neuronWeight / dataLength)))
764         })
765         neuron.updateWeights(neuronWeights)
766       })
767     })
768   }
769 
770   function* alphaUpdaterGen (alpha) {
771     alpha = alpha || 0.5
772     var error
773     var lastError
774     while (true)  {
775       if (error && lastError) {
776         if (error <= lastError) alpha *= progressiveAlpha.creep
777         else alpha *= progressiveAlpha.reversal
778       }
779       if (progressiveAlpha.floor && alpha < progressiveAlpha.floor) alpha = progressiveAlpha.floor
780       lastError = error
781       error = yield alpha
782     }
783   }
784 }

786 function batch (trainer, freq) {
787   var timer = new Date()
788   var next = function (inp) {
789     return new Promise(resolve => {
790       var newTime = new Date()
791       if ((newTime - timer) > freq) {
792         setTimeout(() => {
793           timer = newTime
794           resolve(trainer.next(inp))
795         }, 0)
796       } else {
797         resolve(trainer.next(inp))
798       }
799     })
800   }
801 
802   return { next }
803 }

805 function race (trainerArrayInput, rules, opts) {
806   opts = opts || {}
807   var epoch = 0
808   var trainerArray = trainerArrayInput.slice(0)
809   var lastResults = []
810   var errorThreshold = opts.errorThreshold || 0
811   var maxEpochs = opts.maxEpochs || null
812   var currMinError
813   while ((!currMinError || currMinError > errorThreshold) && (!maxEpochs || epoch <= maxEpochs)) {
814     lastResults = trainerArray.map((trainer, ind) => {
815       return (lastResults[ind] && lastResults[ind].done) ? lastResults[ind] : trainer.gen.next()
816     })
817     currMinError = calcCurrMinError(lastResults)
818     if (rules[epoch]) {
819       console.log(`Trimming to ${rules[epoch]} trainers`)
820       trainerArray = trim(trainerArray, lastResults, rules[epoch])
821     }
822     console.log(`Epoch ${epoch} - current global min error is ${currMinError}`)
823     epoch = epoch + 1
824   }
825 
826   return trainerArray
827 
828   function trim (trainerArray, lastResults, num) {
829     var indices = lastResults.slice(0).sort(results => -results.value.minError).map(results => lastResults.findIndex(theseResults => results.value.error === theseResults.value.error)).slice(0, num)
830     return indices.map(ind => trainerArray[ind])
831   }
832 
833   function calcCurrMinError (lastResults) {
834     return lastResults.reduce((min, results) => (min === null || results.value.error < min) ? results.value.error : min, null)
835   }
836 }

847  * @param  {string} key   transfer Function name
848  * @param  {function(x) => y} fn    The transfer function itself, which should take a number and output a number.  It should be differentiable if it's to be used in network training.
849  * @param  {function(x) => y'} deriv The derivative of the transfer function, which is required for back-propagation.  It should take a number and output a number.

853   transferFunctions[key].derivative = deriv

881 function networkSum(X, Y, ignoreLayers) {
882   return X.map((xLayer, layerInd) => {
883     if (ignoreLayers && ignoreLayers.some(ind => ind === layerInd)) return xLayer
884     return xLayer.map((xNeuron, neuronInd) => {
885       return xNeuron.map((xWeight, weightInd) => {
886         return xWeight + Y[layerInd][neuronInd][weightInd]
887       })
888     })
889   })
890 }
